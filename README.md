/*
	比较运算符，计算结果只有2个可能，true，false
	结算结果的数据类型，boolean类型
	== 比较符号两边的数据
	!= 比较符号两边的数据
	>  比较符号两边的数据
	<  比较符号两边的数据
	>= 比较符号两边的数据
	<= 比较符号两边的数据
*/
public class Operator_3{
	public static void main(String[] args){
		//赋值运算符，将右面的值，赋值给左边的变量
		int i = 3;
		int j = 4;
		System.out.println(i=j);
		System.out.println(i==j);
		System.out.println(2==1);
		System.out.println(2!=1);
	}
}
